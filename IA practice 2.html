bins = [60, 70, 80, 90, 100]
labels = ['60–70', '70–80', '80–90', '90–100']
df['Attendance_Range'] = pd.cut(df['Attendance (%)'], bins=bins, labels=labels, right=True)

attendance_freq = df['Attendance_Range'].value_counts().sort_index()

plt.figure(figsize=(6,4))
plt.bar(attendance_freq.index, attendance_freq.values, color='mediumseagreen')
plt.title('Frequency of Attendance Percentage Ranges')
plt.xlabel('Attendance Range (%)')
plt.ylabel('Number of Students')
plt.grid(axis='y')
plt.show()
